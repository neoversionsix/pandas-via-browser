Excel to HTML Table





packages = ["matplotlib", "pandas"] import pandas as pd import matplotlib.pyplot as plt import io import sys def generateTable(): data = document.getElementById("excel-data").value df = pd.read_csv(io.StringIO(data), sep='\t', header=None) df = df.applymap(lambda x: x*2 if isinstance(x, (int, float)) else x) table = document.getElementById("html-table") table.innerHTML = df.to_html(index=False, header=False) table.style.display = "block"
